50Adaptation::ServiceFilter &Adaptation::ServiceFilter::operator =(const ServiceFilter &f)
51{
52 if (this != &f) {
53 method = f.method;
54 point = f.point;
55 HTTPMSGUNLOCK(request);
56 HTTPMSGUNLOCK(reply);
57 request = f.request;
58 HTTPMSGLOCK(request);
59 reply = f.reply;
60 if (reply)
61 HTTPMSGLOCK(reply);
62 }
63 return *this;
64}
