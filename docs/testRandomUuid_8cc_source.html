52void
53TestRandomUuid::testUniqueness()
54{

56 while (uniqueIds.size() < 1000) {
57 const auto inserted = uniqueIds.insert(ToSBuf(RandomUuid())).second;
58 CPPUNIT_ASSERT_MESSAGE("few generated UUIDs are unique", inserted);
59 }
60}
