56CpuCount(const cpu_set_t *set)
57{
58 int count = 0;
59 for (int i = 0; i < CPU_SETSIZE; ++i) {
60 if (CPU_ISSET(i, set))
61 ++count;
62 }

75 if (CPU_ISSET(i, srcset1) && CPU_ISSET(i, srcset2))
