23void *
24MemPoolMalloc::allocate()
25{
26 void *obj = nullptr;
27 if (!freelist.empty()) {
28 obj = freelist.top();
29 freelist.pop();
30 }
31 if (obj) {
32 --meter.idle;
33 ++countSavedAllocs;
34 if (doZero)
35 (void)VALGRIND_MAKE_MEM_DEFINED(obj, objectSize);
36 else
37 (void)VALGRIND_MAKE_MEM_UNDEFINED(obj, objectSize);
38 } else {
39 if (doZero)
40 obj = xcalloc(1, objectSize);
41 else
42 obj = xmalloc(objectSize);
43 ++meter.alloc;
44 }
45 ++meter.inuse;
46 return obj;
47}

49void
50MemPoolMalloc::deallocate(void *obj)
51{
52 --meter.inuse;
53 if (MemPools::GetInstance().idleLimit() == 0) {
54 xfree(obj);
55 --meter.alloc;
56 } else {
57 if (doZero)
58 memset(obj, 0, objectSize);
59 (void)VALGRIND_MAKE_MEM_NOACCESS(obj, objectSize);
60 ++meter.idle;
61 freelist.push(obj);
62 }
63}
