77void
78Snmp::Forwarder::sendError(int error)
79{
80 debugs(49, 3, MYNAME);
81
82 if (fd < 0)
83 return; // client gone
84
85 Snmp::Request& req = static_cast<Snmp::Request&>(*request);
86 req.pdu.command = SNMP_PDU_RESPONSE;
87 req.pdu.errstat = error;
88 u_char buffer[SNMP_REQUEST_SIZE];
89 int len = sizeof(buffer);
90 if (snmp_build(&req.session, &req.pdu, buffer, &len) == 0)
91 comm_udp_sendto(fd, req.address, buffer, len);
92 else
93 debugs(49, DBG_IMPORTANT, "ERROR: Failed to encode an error response to SNMP agent query from " << req.address);
94}

96void
97Snmp::SendResponse(const Ipc::RequestId requestId, const Pdu &pdu)
98{
99 debugs(49, 5, MYNAME);
100 // snmpAgentResponse() can modify arg
101 Pdu tmp = pdu;
102 Snmp::Response response(requestId);
103 snmp_pdu* response_pdu = nullptr;
104 try {
105 response_pdu = snmpAgentResponse(&tmp);
106 Must(response_pdu != nullptr);
107 response.pdu = static_cast<Pdu&>(*response_pdu);
108 snmp_free_pdu(response_pdu);
109 } catch (const std::exception& e) {
110 debugs(49, DBG_CRITICAL, e.what());
111 response.pdu.command = SNMP_PDU_RESPONSE;
112 response.pdu.errstat = SNMP_ERR_GENERR;
113 }
114 Ipc::TypedMsgHdr message;
115 response.pack(message);
116 Ipc::SendMessage(Ipc::Port::CoordinatorAddr(), message);
117}
