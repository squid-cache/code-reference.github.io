97template <typename S>
98class SuccessSumTester
99{
100public:
101 template <typename... Args>
102 static S Test(Args... args)
103 {
104 // to show every non-overflowing sum to be tested:
105 //std::cout << SumToString<S>(args...) << " = " << +sum << "\n";
106
107 const auto ns = NaturalSum<S>(args...);
108 CPPUNIT_ASSERT_MESSAGE(SumToString<S>(args...) + " does not overflow",
109 ns.has_value());
110
111 const auto sum = ns.value();
112 const auto expected = RawSum<S>(args...);
113 CPPUNIT_ASSERT_MESSAGE(
114 SumToString<S>(args...) + " = " + OperandToString(expected) + " rather than " + OperandToString(sum),
115 sum == expected);
116
117 return sum;
118 }

195void
196TestMath::testNaturalSum()
197{
198 /*
199 * To simplify spelling out of these repetitive test cases, we let function
200 * parameters determine the summation type. Regular code should not do that,
201 * and our public summation APIs do not provide this dangerous shortcut.
202 */
203
204 // negative parameters are banned in any position
205 TestOverflowForEitherSummationType(min64s, zero8s);
206 TestOverflowForEitherSummationType(min64s, zero8u);
207 TestOverflowForEitherSummationType(min64s, max64s);
208 TestOverflowForEitherSummationType(min64s, max64u);
209 TestOverflowForEitherSummationType(min8s, zero8s);
210 TestOverflowForEitherSummationType(min8s, zero8s);
211 TestOverflowForEitherSummationType(min8s, zero8u);
212 TestOverflowForEitherSummationType(min8s, max64s);
213 TestOverflowForEitherSummationType(min8s, max64u);
214 TestOverflowForEitherSummationType(-1, -1);
215 TestOverflowForEitherSummationType(-1, zero8s);
216 TestOverflowForEitherSummationType(-1, zero8u);
217 TestOverflowForEitherSummationType(-1, max64s);
218 TestOverflowForEitherSummationType(-1, max64u);
219
220 // these overflow regardless of which parameter determines the summation type
221 TestOverflowForEitherSummationType(max8u, one8u);
222 TestOverflowForEitherSummationType(max8u, one8s);
223 TestOverflowForEitherSummationType(max8u, max8s);
224 TestOverflowForEitherSummationType(max8s, one8s);
225 TestOverflowForEitherSummationType(max64u, one8u);
226 TestOverflowForEitherSummationType(max64u, one8s);
227 TestOverflowForEitherSummationType(max64u, one64u);
228 TestOverflowForEitherSummationType(max64u, one64s);
229 TestOverflowForEitherSummationType(max64u, max64s);
230 TestOverflowForEitherSummationType(max64s, one8u);
231 TestOverflowForEitherSummationType(max64s, one8s);
232 TestOverflowForEitherSummationType(max64s, one64s);
233
234 // these overflow only if the second parameter determines the summation type
235 TestSuccessForFirstSummationType(one8u, max8s);
236 TestSuccessForFirstSummationType(one64u, max8u);
237 TestSuccessForFirstSummationType(one64u, max64s);
238 TestSuccessForFirstSummationType(one64s, max8u);
239 TestSuccessForFirstSummationType(one64s, max8s);
240 TestSuccessForFirstSummationType(max64u, zero8u);
241 TestSuccessForFirstSummationType(max64u, zero8s);
242 TestSuccessForFirstSummationType(max64s, zero8u);
243 TestSuccessForFirstSummationType(max64s, zero8s);
244
245 // a few sums with known values
246 CPPUNIT_ASSERT_EQUAL(zero8s, GoodSum(zero8s, zero8u));
247 CPPUNIT_ASSERT_EQUAL(zero64s, GoodSum(zero64s, zero64u));
248 CPPUNIT_ASSERT_EQUAL(2, GoodSum(1, 1));
249 CPPUNIT_ASSERT_EQUAL(uint64_t(2), GoodSum(one64u, one64s));
250 CPPUNIT_ASSERT_EQUAL(6u, GoodSum(1u, 2, 3));
251 CPPUNIT_ASSERT_EQUAL(max64u, GoodSum(zero64u, max64u));
252 CPPUNIT_ASSERT_EQUAL(max64s, GoodSum(zero64s, max64s));
253 CPPUNIT_ASSERT_EQUAL(one64u + max64s, GoodSum(one64u, max64s));
254 CPPUNIT_ASSERT_EQUAL(max64u, GoodSum(max64u, zero8s));
255 CPPUNIT_ASSERT_EQUAL(max64s, GoodSum(max64s, zero8s));
256
257 // long argument lists (odd and even lengths)
258 CPPUNIT_ASSERT_EQUAL(15, NaturalSum<int>(1, 2, 3, 4, 5).value());
259 CPPUNIT_ASSERT_EQUAL(21, NaturalSum<int>(1, 2, 3, 4, 5, 6).value());
260
261 // test SetToNaturalSumOrMax() when the sum is too big for the variable
262 long expires = 0;
263 const auto result = SetToNaturalSumOrMax(expires, max64u, zero8u);
264 CPPUNIT_ASSERT_EQUAL(std::numeric_limits<long>::max(), expires);
265 CPPUNIT_ASSERT_EQUAL(expires, result);
266}
